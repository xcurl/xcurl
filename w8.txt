    <div id="controls">
        <button onclick="show((i-1+n)%n)">Prev</button>
        <button onclick="show((i+1)%n)">Next</button>
    </div>

    <script>
        let imgs = document.querySelectorAll("#slider img");
        let n = imgs.length;
        let i=0;

        function show(x){
            imgs.forEach(imgs=>imgs.classList.remove("active"));
            i=x;
            imgs[i].classList.add("active");
        }
        setInterval(function(){show((i+1)%n);},3000);
    </script>
