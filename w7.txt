    <script>
        let current ="";
        function topit(id){
            if(current && current!==id){
                document.getElementById(current).style.zIndex = "";
                //current was passed as string 
            }
            document.getElementById(id).style.zIndex = "20";
            current=id;
        }
        function reset(){
            if(current){
                document.getElementById(current).style.zIndex = "";
                current="";
            }
        }
